        let videoSkipped = false;
        let videoSkipTimer = null;
        let videoDurationTimer = null;
        const VIDEO_DURATION = 39; // 39 seconds

        function showVideoModal() {
            const videoModal = document.getElementById("videoModal");
            const video = document.getElementById("engagementVideo");
            
            videoSkipped = false;
            
            if (videoModal) {
                videoModal.classList.add("showVideo");
                
                if (video) {
                    // Reset video
                    video.currentTime = 0;
                    video.muted = false; // Keep muted for autoplay reliability
                    
                    // Remove all previous event listeners
                    video.removeEventListener('ended', videoEnded);
                    video.removeEventListener('ended', handlePiPEnd);
                    
                    // Start video
                    const playPromise = video.play();
                    if (playPromise !== undefined) {
                        playPromise.catch(error => {
                            console.log("Video play error:", error);
                        });
                    }
                    
                    // Add event listener for video end
                    video.addEventListener('ended', videoEnded, { once: true });
                    
                    // Auto-close video and show popup after 39 seconds
                    startVideoDurationTimer();
                    
                    console.log("Video modal shown, video started");
                }
            }
        }

        function startVideoDurationTimer() {
            if (videoDurationTimer) clearInterval(videoDurationTimer);
            
            let elapsed = 0;
            videoDurationTimer = setInterval(() => {
                elapsed++;
                if (elapsed >= VIDEO_DURATION) {
                    clearInterval(videoDurationTimer);
                    videoEnded();
                }
            }, 1000);
        }

        function skipVideo() {
            const video = document.getElementById("engagementVideo");
            const videoModal = document.getElementById("videoModal");
            
            if (!video || !videoModal) return;

            videoSkipped = true;
            if (videoDurationTimer) clearInterval(videoDurationTimer);

            // Remove all event listeners to prevent double trigger
            video.removeEventListener('ended', videoEnded);
            video.removeEventListener('ended', handlePiPEnd);

            // Hide the main video modal
            videoModal.classList.remove("showVideo");

            console.log("Video skipped, attempting PiP mode");

            // Try Picture-in-Picture mode
            if (document.pictureInPictureElement) {
                // Already in PiP, just continue
                console.log("Already in PiP mode");
                video.play();
                video.addEventListener('ended', handlePiPEnd, { once: true });
                startVideoDurationTimer();
            } else if (video.requestPictureInPicture) {
                video.requestPictureInPicture()
                    .then(() => {
                        console.log("Entered Picture-in-Picture mode");
                        video.play();
                        video.addEventListener('ended', handlePiPEnd, { once: true });
                        startVideoDurationTimer();
                    })
                    .catch((error) => {
                        console.log("PiP request failed:", error);
                        // Fallback: show instagram popup
                        showInstagramPopup();
                    });
            } else {
                console.log("PiP not supported on this browser");
                // Fallback: show instagram popup
                showInstagramPopup();
            }
        }

        function handlePiPEnd() {
            console.log("Video ended in PiP mode");
            videoEnded();
        }

        function closeVideoModal() {
            const videoModal = document.getElementById("videoModal");
            const video = document.getElementById("engagementVideo");

                videoModal.classList.remove("showVideo");

        // low-power detection helper
        function isLowPower() {
            try {
                if (window.matchMedia && window.matchMedia('(prefers-reduced-motion: reduce)').matches) return true;
                const conn = navigator.connection || navigator.mozConnection || navigator.webkitConnection;
                if (conn && conn.effectiveType) {
                    const slow = ['slow-2g', '2g', '3g'];
                    if (slow.includes(conn.effectiveType)) return true;
                }
                if (navigator.deviceMemory && navigator.deviceMemory < 1.5) return true;
                return false;
            } catch (e) {
                return false;

            function addParticle(x, y, opts = {}) {

                particles.push(Object.assign({

                }

            function drawHeart(ctx, x, y, size, color, alpha) {
                ctx.save();
                ctx.translate(x, y);

                const heartChance = low ? 0.08 : 0.25;
                const delay = low ? 500 : 350;

                for (let i = 0; i < bursts; i++) {
                    setTimeout(() => {
                        const x = (i + Math.random() * (low ? 0.9 : 0.6)) / bursts * canvas.width;
                        const y = Math.random() * canvas.height * (low ? 0.5 : 0.6) + 20;
                        addBurst(x, y, perBurst, heartChance);
                    }, i * delay);
                }

                // run animation
                requestAnimationFrame(loop);
            }

            function loop() {
                if (!running && particles.length === 0) return;
                if (document.hidden) return; // pause when tab hidden

                ctx.clearRect(0, 0, canvas.width, canvas.height);

                for (let i = particles.length - 1; i >= 0; i--) {
                    const p = particles[i];
                    p.x += p.vx;
                    p.y += p.vy;
                    p.vy += 0.03; // gentle gravity
                    p.age++;
                    const alpha = Math.max(0, 1 - p.age / p.life);
                    if (alpha <= 0) { particles.splice(i, 1); continue; }

                    if (p.type === 'heart') {
                        drawHeart(ctx, p.x, p.y, p.size, p.color, alpha);
                    } else {
                        ctx.beginPath();
                        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
                        ctx.fillStyle = `hsla(${p.color},90%,60%,${alpha})`;
                        ctx.fill();
                    }
                }

                if (particles.length > 0) {
                    requestAnimationFrame(loop);
                } else {
                    running = false;
                }
            }

            // expose functions globally
            window.startFullSprinkle = startFullSprinkle;
            window.startFireworks = startFullSprinkle;
        })();

        /* Circular slider: shift-first technique to create infinite loop without gaps */
        function initCircularSlider() {
            const slider = document.querySelector('.slider');
            const slides = document.querySelector('.slides');
            if (!slider || !slides) return;

            let isMoving = false;
            let interval = 3000;

            function moveOnce() {
                if (isMoving) return;
                const first = slides.firstElementChild;
                if (!first) return;

                const slideWidth = slider.clientWidth;
                isMoving = true;
                slides.style.transition = 'transform 600ms ease';
                slides.style.transform = `translateX(-${slideWidth}px)`;

                const onEnd = () => {
                    slides.removeEventListener('transitionend', onEnd);
                    slides.style.transition = 'none';
                    slides.appendChild(first);
                    slides.style.transform = 'translateX(0)';
                    // small timeout to re-enable transitions for the next move
                    setTimeout(() => {
                        slides.style.transition = '';
                        isMoving = false;
                    }, 20);
                };

                slides.addEventListener('transitionend', onEnd);
            }

            let timer = setInterval(moveOnce, interval);

            // stop on hover to be friendly, resume on leave
            slider.addEventListener('mouseenter', () => clearInterval(timer));
            slider.addEventListener('mouseleave', () => timer = setInterval(moveOnce, interval));

            // handle resize to ensure correct width
            window.addEventListener('resize', () => {
                // no-op: widths are recalculated on each move
            });
        }

        document.addEventListener('DOMContentLoaded', initCircularSlider);

        /* SERVICE WORKER - CACHING */
        if ('serviceWorker' in navigator) {
            navigator.serviceWorker.register('sw.js').then(reg => {
                console.log('Service Worker registered successfully:', reg);
            }).catch(err => {
                console.log('Service Worker registration failed:', err);
            });
        }
